    updateDisplay();
}

    display.textContent = currentValue;
}

### ƒÜwiczenie ko≈Ñcowe - Mini projekt

Zanim przejdziesz do DOM, spr√≥buj stworzyƒá:

```javascript
// System zarzƒÖdzania studentami (bez DOM, tylko console)

const students = [];

function addStudent(name, age, grades) {
    students.push({
        id: Date.now(),
        name: name,
        age: age,
        grades: grades
    });
}

function getStudentById(id) {
    return students.find(student => student.id === id);
}

function getAverageGrade(studentId) {
    const student = getStudentById(studentId);
    if (!student) return null;
    
    const sum = student.grades.reduce((acc, grade) => acc + grade, 0);
    return sum / student.grades.length;
}

function getAllStudents() {
    return students.map(student => ({
        ...student,
        average: getAverageGrade(student.id)
    }));
}

function getTopStudents(count = 3) {
    return getAllStudents()
        .sort((a, b) => b.average - a.average)
        .slice(0, count);
}

// Testowanie
addStudent('Jan', 20, [4, 5, 3, 5]);
addStudent('Anna', 22, [5, 5, 4, 5]);
addStudent('Piotr', 21, [3, 4, 3, 4]);

console.log('Wszyscy studenci:', getAllStudents());
console.log('Top 2 student√≥w:', getTopStudents(2));
```

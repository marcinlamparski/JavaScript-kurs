# Kurs Podstaw JavaScript - Lekcje 7-8

## Lekcja 7: Instrukcje warunkowe

### Instrukcja if

Instrukcje warunkowe pozwalają na wykonanie kodu tylko wtedy, gdy spełniony jest określony warunek.

```javascript
let wiek = 20;

if (wiek >= 18) {
    console.log("Jesteś pełnoletni");
}

// Przykład z operatorem logicznym
let temperatura = 25;

if (temperatura > 20 && temperatura < 30) {
    console.log("Idealna pogoda!");
}
```

### Instrukcja if...else

```javascript
let punkty = 75;

if (punkty >= 60) {
    console.log("Zdałeś egzamin!");
} else {
    console.log("Niestety nie zdałeś.");
}

// Praktyczny przykład
let godzina = 14;

if (godzina < 12) {
    console.log("Dzień dobry!");
} else {
    console.log("Dobry wieczór!");
}
```

### Instrukcja if...else if...else

```javascript
let ocena = 4;

if (ocena === 5) {
    console.log("Celujący!");
} else if (ocena === 4) {
    console.log("Dobry");
} else if (ocena === 3) {
    console.log("Dostateczny");
} else if (ocena === 2) {
    console.log("Dopuszczający");
} else {
    console.log("Niedostateczny");
}

// Przykład z zakresami
let wynik = 85;

if (wynik >= 90) {
    console.log("Ocena: A");
} else if (wynik >= 80) {
    console.log("Ocena: B");
} else if (wynik >= 70) {
    console.log("Ocena: C");
} else if (wynik >= 60) {
    console.log("Ocena: D");
} else {
    console.log("Ocena: F");
}
```

### Operator warunkowy (ternary)

Skrócona forma if...else dla prostych warunków.

```javascript
// Składnia: warunek ? wartośćJeśliTrue : wartośćJeśliFalse

let wiek = 20;
let status = wiek >= 18 ? "pełnoletni" : "niepełnoletni";
console.log(status); // "pełnoletni"

// Zagnieżdżony operator ternary (nie nadużywaj!)
let punkty = 75;
let ocena = punkty >= 90 ? "A" : punkty >= 80 ? "B" : punkty >= 70 ? "C" : "D";
console.log(ocena); // "C"

// Praktyczny przykład
let temperatura = 15;
console.log(`Pogoda jest ${temperatura > 20 ? "ciepła" : "chłodna"}`);
```

### Instrukcja switch

Alternatywa dla wielu if...else, gdy sprawdzamy jedną wartość.

```javascript
let dzienTygodnia = 3;
let nazwaDnia;

switch (dzienTygodnia) {
    case 1:
        nazwaDnia = "Poniedziałek";
        break;
    case 2:
        nazwaDnia = "Wtorek";
        break;
    case 3:
        nazwaDnia = "Środa";
        break;
    case 4:
        nazwaDnia = "Czwartek";
        break;
    case 5:
        nazwaDnia = "Piątek";
        break;
    case 6:
        nazwaDnia = "Sobota";
        break;
    case 7:
        nazwaDnia = "Niedziela";
        break;
    default:
        nazwaDnia = "Nieprawidłowy dzień";
}

console.log(nazwaDnia); // "Środa"
```

**Ważne:** Nie zapomnij o `break`! Bez niego kod będzie wykonywać następne case'y.

#### Switch - przypadki grupowe

```javascript
let miesiac = 2;
let poraPoku;

switch (miesiac) {
    case 12:
    case 1:
    case 2:
        poraPoku = "Zima";
        break;
    case 3:
    case 4:
    case 5:
        poraPoku = "Wiosna";
        break;
    case 6:
    case 7:
    case 8:
        poraPoku = "Lato";
        break;
    case 9:
    case 10:
    case 11:
        poraPoku = "Jesień";
        break;
    default:
        poraPoku = "Nieprawidłowy miesiąc";
}

console.log(poraPoku); // "Zima"
```

### Wartości truthy i falsy

JavaScript automatycznie konwertuje wartości na boolean w kontekście warunkowym.

```javascript
// Wartości FALSY (traktowane jako false):
// false, 0, "" (pusty string), null, undefined, NaN

if (0) {
    console.log("To się nie wykona");
}

if ("") {
    console.log("To też się nie wykona");
}

// Wartości TRUTHY (wszystko inne, w tym):
// true, liczby różne od 0, niepuste stringi, tablice, obiekty

if (1) {
    console.log("To się wykona"); // ✓
}

if ("tekst") {
    console.log("To też się wykona"); // ✓
}

if ([]) {
    console.log("Pusta tablica też jest truthy!"); // ✓
}

// Praktyczne zastosowanie
let userName = "";

if (userName) {
    console.log(`Witaj, ${userName}!`);
} else {
    console.log("Proszę podać imię");
}
```

### Skrócone warunki logiczne

```javascript
// Logical OR (||) - zwraca pierwszą truthy wartość
let userName = "";
let displayName = userName || "Gość";
console.log(displayName); // "Gość"

let points = 0;
let finalPoints = points || 100;
console.log(finalPoints); // 100 (0 to falsy)

// Logical AND (&&) - zwraca pierwszą falsy lub ostatnią truthy
let isLoggedIn = true;
let hasPermission = true;

isLoggedIn && hasPermission && console.log("Dostęp przyznany");

// Nullish coalescing operator (??) - ES2020
let score = 0;
let result1 = score || 100; // 100 (0 to falsy)
let result2 = score ?? 100; // 0 (tylko null i undefined są zastępowane)
console.log(result1, result2); // 100, 0
```

## Lekcja 8: Pętle (Loops)

### Pętla for

Najczęściej używana pętla, gdy znamy liczbę iteracji.

```javascript
// Podstawowa składnia: for (inicjalizacja; warunek; aktualizacja)

for (let i = 0; i < 5; i++) {
    console.log(i);
}
// Wynik: 0 1 2 3 4

// Zliczanie wstecz
for (let i = 5; i > 0; i--) {
    console.log(i);
}
// Wynik: 5 4 3 2 1

// Krok większy niż 1
for (let i = 0; i <= 10; i += 2) {
    console.log(i);
}
// Wynik: 0 2 4 6 8 10
```

#### For z tablicami

```javascript
let owoce = ["jabłko", "banan", "gruszka", "pomarańcza"];

for (let i = 0; i < owoce.length; i++) {
    console.log(`Owoc ${i + 1}: ${owoce[i]}`);
}

// Iteracja wsteczna
for (let i = owoce.length - 1; i >= 0; i--) {
    console.log(owoce[i]);
}
```

#### Zagnieżdżone pętle

```javascript
// Tabliczka mnożenia
for (let i = 1; i <= 3; i++) {
    for (let j = 1; j <= 3; j++) {
        console.log(`${i} x ${j} = ${i * j}`);
    }
}

// Tworzenie macierzy
let macierz = [];
for (let i = 0; i < 3; i++) {
    macierz[i] = [];
    for (let j = 0; j < 3; j++) {
        macierz[i][j] = i * 3 + j;
    }
}
console.log(macierz); // [[0,1,2], [3,4,5], [6,7,8]]
```

### Pętla while

Wykonuje kod dopóki warunek jest prawdziwy.

```javascript
let licznik = 0;

while (licznik < 5) {
    console.log(licznik);
    licznik++;
}
// Wynik: 0 1 2 3 4

// Praktyczny przykład
let suma = 0;
let liczba = 1;

while (suma < 100) {
    suma += liczba;
    liczba++;
}
console.log(`Suma: ${suma}, Liczba: ${liczba}`);
```

**Uwaga:** Uważaj na nieskończone pętle!

```javascript
// BŁĄD - nieskończona pętla:
// let i = 0;
// while (i < 5) {
//     console.log(i);
//     // Brak i++, więc warunek zawsze true
// }
```

### Pętla do...while

Wykonuje kod przynajmniej raz, a potem sprawdza warunek.

```javascript
let licznik = 0;

do {
    console.log(licznik);
    licznik++;
} while (licznik < 5);
// Wynik: 0 1 2 3 4

// Różnica: kod wykona się przynajmniej raz
let x = 10;

do {
    console.log("To się wykona mimo że x >= 10");
} while (x < 10);
```

### Pętla for...of (dla tablic)

Nowoczesna pętla do iteracji po tablicach.

```javascript
let owoce = ["jabłko", "banan", "gruszka"];

for (let owoc of owoce) {
    console.log(owoc);
}
// Wynik: jabłko, banan, gruszka

// Z indeksem (użyj entries())
for (let [indeks, owoc] of owoce.entries()) {
    console.log(`${indeks}: ${owoc}`);
}

// Dla stringów
let tekst = "Hello";
for (let litera of tekst) {
    console.log(litera);
}
// Wynik: H e l l o
```

### Pętla for...in (dla obiektów)

Iteruje po kluczach obiektu (poznamy to dokładniej w następnej lekcji).

```javascript
let osoba = {
    imie: "Jan",
    wiek: 30,
    miasto: "Warszawa"
};

for (let klucz in osoba) {
    console.log(`${klucz}: ${osoba[klucz]}`);
}
// Wynik:
// imie: Jan
// wiek: 30
// miasto: Warszawa
```

### Instrukcje break i continue

```javascript
// break - przerywa pętlę
for (let i = 0; i < 10; i++) {
    if (i === 5) {
        break; // Kończy pętlę
    }
    console.log(i);
}
// Wynik: 0 1 2 3 4

// continue - pomija bieżącą iterację
for (let i = 0; i < 10; i++) {
    if (i % 2 === 0) {
        continue; // Pomija parzyste liczby
    }
    console.log(i);
}
// Wynik: 1 3 5 7 9

// Praktyczny przykład - wyszukiwanie w tablicy
let liczby = [10, 23, 5, 78, 34, 12];
let szukana = 78;
let znaleziono = false;

for (let liczba of liczby) {
    if (liczba === szukana) {
        znaleziono = true;
        break; // Znaleźliśmy, koniec szukania
    }
}

console.log(znaleziono ? "Znaleziono!" : "Nie znaleziono");
```

## Podsumowanie Lekcji 7-8

### Poznałeś:
✅ Instrukcje warunkowe: if, else if, else  
✅ Operator warunkowy (ternary)  
✅ Instrukcję switch  
✅ Wartości truthy i falsy  
✅ Pętle: for, while, do...while  
✅ Pętle specjalne: for...of, for...in  
✅ Instrukcje break i continue  

### Najważniejsze zasady:
- Zawsze używaj `===` zamiast `==` w warunkach
- `switch` wymaga `break` w każdym case
- Wybieraj odpowiednią pętlę: `for` gdy znasz liczbę iteracji, `while` gdy nie
- `for...of` to najlepszy sposób iteracji po tablicach
- Uważaj na nieskończone pętle!

### Typowe błędy:
❌ Zapomnienie `break` w switch  
❌ Użycie `=` zamiast `===` w warunku  
❌ Nieskończona pętla (brak inkrementacji licznika)  
❌ Off-by-one error (błąd o 1) w pętlach
